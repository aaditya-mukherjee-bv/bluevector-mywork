        }

    function processReviews(data) {
      console.log("[processReviews] Input data:", data);
      let resultData;
      if (typeof data.result === 'string') {
        resultData = JSON.parse(data.result);
      }
      else if (Array.isArray(data.result)) {
        resultData = data.result;
      }
      else if (Array.isArray(data)) {
        resultData = data;
      }
      else {
        console.error("[processReviews] Unexpected data format:", data);
        throw new Error('Unexpected data format');
      }
      window.reviews = resultData.map((item, index) => {
        console.log("[processReviews] Mapping item:", item);
        return {
          id: item.id || `review-${index}
`,
          title: item.title || 'Untitled Review',
          body: item.body || 'No review content available',
          likes: item.likes ?? 0,
          user_id: item.user_id || 'anonymous',
          user_name: item.user_name || 'Anonymous User',
          user_picture: item.user_picture || null,
          listing_id: item.listing_id || null,
          rating: item.rating ?? 0,
          status: item.status || 'Pending',
          created_on: item.created_on ? new Date(item.created_on) : new Date(),
        };
      }
                                     );
      console.log("[processReviews] Parsed reviews:", window.reviews);
      // Calculate and store the original average
      if (window.reviews.length > 0) {
        let totalRating = 0;
        window.reviews.forEach(review => {
          totalRating += review.rating || 5;
        }
                              );
        originalAverageRating = (totalRating / window.reviews.length).toFixed(2);
      }
      else {
        originalAverageRating = null;

      }

        loader.style.display = '';
        loader.classList.add('d-none');
        if (timerId) clearTimeout(timerId);
      }

    function addReadMore(selector, maxChars = 100) {
      const el = document.querySelector(selector);
      if (!el) {
        return;
      }
      const fullText = el.textContent.trim();
      if (fullText.length <= maxChars) {
        return;
      }
      const shortText = fullText.slice(0, maxChars) + '... ';
      el.textContent = shortText;
      const moreLink = document.createElement('a');
      moreLink.href = '#';
      moreLink.textContent = 'Read more';
      moreLink.style.color = '#2563eb';
      moreLink.style.cursor = 'pointer';
      moreLink.addEventListener('click', function (e) {
        e.preventDefault();
        if (moreLink.textContent === 'Read more') {
          el.textContent = fullText + ' ';
          moreLink.textContent = 'Read less';
          el.appendChild(moreLink);
        }
        else {
          el.textContent = shortText;
          moreLink.textContent = 'Read more';
          el.appendChild(moreLink);
        }
      }
                               );
      el.appendChild(moreLink);
      console.log('addReadMore: Read more link added');
    }

    function renderBadges(inputId, containerId, badgeClass, type, useIcon = false, consise = true) {
      const raw = (document.getElementById(inputId)?.value || '');
      const categories = raw
      .replace(/[{}]/g, '')
      .split(',')
      .map(cat => cat?.trim().replace(/^'+|'+$/g, ''))
      .filter(cat => cat && cat.length > 0 && !['none', 'null', '/'].includes(cat.toLowerCase()));
      if (categories.length === 0) {
        const container = document.getElementById(containerId);
        if (container) container.style.display = 'none';
        return;
      }
      const container = document.getElementById(containerId);
      container.innerHTML = '';
      const maxShow = consise ? 6 : categories.length;
      function render(showAll) {
        container.innerHTML = '';
        const itemsToRender = showAll ? categories : categories.slice(0, maxShow);
        itemsToRender.forEach(category => {
          const link = document.createElement('a');
          link.className = badgeClass;
          // Construct the dynamic URL. Note that 'app=f82788c1-b9b2-48b6-81e4-e331ea2582f4' is already part of the base URL.
          // We just need to add the new parameters with an ampersand (&).
          let url = `https://snapapp.com/page/snapapp-solution-center?${type}
=${encodeURIComponent(category)}
`;
          link.href = url;
          if (useIcon && iconMap[category]) {
            link.innerHTML = iconMap[category] + ' ' + category;
          }
          else {
            link.textContent = category;
          }
          container.appendChild(link);
        }
                             );
        if (!showAll && consise && categories.length > maxShow) {
          const moreCount = categories.length - maxShow;
          const moreBtn = document.createElement('span');
          moreBtn.className = badgeClass + ' badge-more';
          moreBtn.style.cursor = 'pointer';
          moreBtn.textContent = `+${moreCount}
`;
          moreBtn.onclick = function () {
            render(true);
          };
          container.appendChild(moreBtn);
        }
        else if (showAll && consise && categories.length > maxShow) {
          const lessBtn = document.createElement('span');
          lessBtn.className = badgeClass + ' badge-more';
          lessBtn.style.cursor = 'pointer';
          lessBtn.textContent = `Show less`;
          lessBtn.onclick = function () {
            render(false);
          };
          container.appendChild(lessBtn);
        }
      }
      render(false);
    }

      const tabTrigger = document.querySelector(`.tab-trigger[data-target="${tabName}

      if (tabTrigger) {
        tabTrigger.classList.add('active');
      }
    }

    async function loadFAQs() {
      const faqList = document.getElementById('faq-list');
      const actionContainer = document.getElementById('action-container');
      if (!faqList || !actionContainer) return;
      // Initially hide the action container
      actionContainer.classList.add('hidden');

      faqList.innerHTML = `
<div class="faq-loading">
<div class="loading-spinner"></div>
<p>Loading FAQs...</p>
    </div>
`;
      try {
        // Get listing ID
        const listingId = document.getElementById('ListingId')?.value;
        if (!listingId) {
          throw new Error('No listing ID found');
        }
        // Fetch FAQ data from backend
        const faqData = await fetchFAQData(listingId);
        if (!faqData || !Array.isArray(faqData) || faqData.length === 0) {
          showNoFAQsMessage();
          return;
        }
        // Render the FAQs
        renderFAQs(faqData);
      }
      catch (error) {
        console.error('Error loading FAQs:', error);
        showErrorMessage('Failed to load FAQs. Please try again.');
      }
    }
    // Fetch FAQ data from the backend
    async function fetchFAQData(listingId) {
      try {
        const payload = {
          expression_field: "invalid_error",
          field_id: "7ce6161a-0cc1-41c9-9a5b-d9cb19af9b1d",
          fields: {
            id: listingId
          }
        };
        const response = await fetch('/preview-expression', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json'
          }
          ,
          body: JSON.stringify(payload)
        }
                                    );
        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}
`);
        }
        const result = await response.json();
        // Handle different response formats
        let faqData;
        if (result.result) {
          if (typeof result.result === 'string') {
            try {
              faqData = JSON.parse(result.result);
            }
            catch (e) {
              faqData = result.result;
            }
          }
          else {
            faqData = result.result;
          }
        }
        else if (result && typeof result === 'string') {
          try {
            faqData = JSON.parse(result);
          }
          catch (e) {
            faqData = result;
          }
        }
        else {
          faqData = result;
        }
        // Validate the data structure
        if (!faqData || faqData === 'None' || faqData === false) {
          return null;
        }
        // Ensure it's an array
        if (!Array.isArray(faqData)) {
          return null;
        }
        return faqData;
      }
      catch (error) {
        console.error('Error fetching FAQ data:', error);
        return null;
      }
    }

    function renderFAQs(faqData) {
      const faqList = document.getElementById('faq-list');
      const actionContainer = document.getElementById('action-container');
      if (!faqList) return;
      // Clear existing content
      faqList.innerHTML = '';
      // Hide action container when FAQs exist
      if (actionContainer) {
        actionContainer.classList.add('hidden');
      }
      // Sort by sequence if available
      const sortedFAQs = [...faqData].sort((a, b) => {
        const seqA = a.sequence || a.id || 0;
        const seqB = b.sequence || b.id || 0;
        return seqA - seqB;
      }
                                          );
      // Create FAQ items
      sortedFAQs.forEach((faq, index) => {
        const faqItem = document.createElement('div');
        faqItem.className = 'faq-item';
        // Use the exact field names from your JSON
        const question = faq.question || faq.title || `Question ${index + 1}
`;
        const answer = faq.answer || faq.content || faq.description || 'No answer available';
        const sequence = faq.sequence || index + 1;
        faqItem.innerHTML = `
<div class="faq-question">
<span class="faq-number">Q${sequence}
    </span>
<h3>${escapeHtml(question)}
    </h3>
<span class="faq-toggle">+</span>
    </div>
<div class="faq-answer">
<p>${escapeHtml(answer)}
    </p>
    </div>
`;
        // Add click event for toggle
        const questionElement = faqItem.querySelector('.faq-question');
        const answerElement = faqItem.querySelector('.faq-answer');
        const toggleElement = faqItem.querySelector('.faq-toggle');
        questionElement.addEventListener('click', function () {
          const isOpen = answerElement.classList.contains('active');
          // Toggle this FAQ
          if (!isOpen) {
            // Close any other open FAQ
            document.querySelectorAll('.faq-answer.active').forEach(el => {
              el.classList.remove('active');
              el.previousElementSibling.querySelector('.faq-toggle').textContent = '+';
            }
                                                                   );
            answerElement.classList.add('active');
            toggleElement.textContent = '';
          }
          else {
            answerElement.classList.remove('active');
            toggleElement.textContent = '+';
          }
        }
                                        );
        faqList.appendChild(faqItem);
      }
                        );
      // Add animation to each FAQ item

          item.style.animationDelay = `${i * 0.1}
s`;
          item.classList.add('animate-in');
        }

      }
                 , 100);
